\subsection{1. ProposerVM Analysis}\label{1-proposervm-analysis}

\subsubsection{1.1 Block Timing and
Windowing}\label{11-block-timing-and-windowing}

\textbf{Files}: \texttt{proposer/windower.go}, \texttt{vm.go}

\textbf{Design}:

\begin{itemize}
\tightlist
\item
  Window duration: 5 seconds per slot
\item
  Max verify windows: 6 (30 seconds)
\item
  Max build windows: 60 (5 minutes)
\item
  Max look-ahead slots: 720 (1 hour)
\end{itemize}

\textbf{Security Observations}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Time Skew Protection} (GOOD)

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{const}\NormalTok{ maxSkew }\OperatorTok{=} \DecValTok{10} \OperatorTok{*}\NormalTok{ time}\OperatorTok{.}\NormalTok{Second}
\end{Highlighting}
\end{Shaded}

  \begin{itemize}
  \tightlist
  \item
    Blocks more than 10 seconds in the future are rejected
  \item
    Prevents time manipulation attacks
  \end{itemize}
\item
  \textbf{Minimum Block Delay} (GOOD)

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{DefaultMinBlockDelay }\OperatorTok{=}\NormalTok{ time}\OperatorTok{.}\NormalTok{Second}
\end{Highlighting}
\end{Shaded}

  \begin{itemize}
  \tightlist
  \item
    Prevents rapid block spam
  \item
    Configurable per network
  \end{itemize}
\item
  \textbf{Monotonic Timestamp Enforcement} (GOOD)

\begin{Shaded}
\begin{Highlighting}[]
\ControlFlowTok{if}\NormalTok{ childTimestamp}\OperatorTok{.}\NormalTok{Before}\OperatorTok{(}\NormalTok{parentTimestamp}\OperatorTok{)} \OperatorTok{\{}
    \ControlFlowTok{return}\NormalTok{ errTimeNotMonotonic}
\OperatorTok{\}}
\end{Highlighting}
\end{Shaded}
\end{enumerate}

\textbf{Potential Issues}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{P-Chain Height Dependency} (LOW RISK)

  \begin{itemize}
  \tightlist
  \item
    Block validation depends on P-Chain height being available
  \item
    During bootstrapping, P-Chain may lag behind
  \item
    Mitigated: Returns \texttt{(time.Time\{\},\ false,\ nil)} allowing
    graceful degradation
  \end{itemize}
\end{enumerate}

\subsubsection{1.2 Proposer Selection}\label{12-proposer-selection}

\textbf{Files}: \texttt{proposer/windower.go},
\texttt{proposer/validators.go}

\textbf{Algorithm}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Fetch validator set from P-Chain at specified height
\item
  Sort validators by NodeID (canonical ordering)
\item
  Use MT19937\_64 PRNG seeded with
  \texttt{chainSource\ \^{}\ blockHeight\ \^{}\ reversedSlot}
\item
  Sample weighted validators without replacement
\end{enumerate}

\textbf{Security Observations}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Deterministic Sampling} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Uses Mersenne Twister with deterministic seed
  \item
    All validators agree on proposer for any slot
  \item
    Slot reversal prevents seed collision between height/slot
    combinations
  \end{itemize}
\item
  \textbf{Weight-Based Selection} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Validators selected proportional to stake weight
  \item
    Uses \texttt{sampler.NewDeterministicWeightedWithoutReplacement}
  \end{itemize}
\item
  \textbf{Empty Validator Set Handling} (GOOD)

\begin{Shaded}
\begin{Highlighting}[]
\ControlFlowTok{if} \BuiltInTok{len}\OperatorTok{(}\NormalTok{validators}\OperatorTok{)} \OperatorTok{==} \DecValTok{0} \OperatorTok{\{}
    \ControlFlowTok{return}\NormalTok{ ids}\OperatorTok{.}\NormalTok{EmptyNodeID}\OperatorTok{,}\NormalTok{ ErrAnyoneCanPropose}
\OperatorTok{\}}
\end{Highlighting}
\end{Shaded}

  \begin{itemize}
  \tightlist
  \item
    When no validators, anyone can propose unsigned blocks
  \item
    Prevents chain halt during validator set transitions
  \end{itemize}
\end{enumerate}

\textbf{Potential Issue}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Validator Grinding} (LOW RISK)

  \begin{itemize}
  \tightlist
  \item
    Seed includes only
    \texttt{chainSource\ \^{}\ blockHeight\ \^{}\ slot}
  \item
    An attacker controlling timing could potentially wait for favorable
    slots
  \item
    Mitigated: Weight-based selection means most slots go to high-stake
    validators
  \end{itemize}
\end{enumerate}

\subsection{2. EVM Integration}\label{2-evm-integration}

\subsubsection{2.1 State Sync}\label{21-state-sync}

\textbf{Files}: \texttt{sync/statesync/state\_syncer.go}

\textbf{Design}:

\begin{itemize}
\tightlist
\item
  Parallel trie syncing with segment splitting
\item
  Code syncer runs asynchronously
\item
  Maximum 8 concurrent workers
\item
  Segment threshold: 500,000 leaves
\end{itemize}

\textbf{Security Observations}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{State Verification} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Syncs against root hash commitment
  \item
    Validates each trie segment against Merkle proofs
  \end{itemize}
\item
  \textbf{Progress Tracking} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Persistent queue survives restarts
  \item
    \texttt{clearIfRootDoesNotMatch} prevents stale state corruption
  \end{itemize}
\end{enumerate}

\textbf{Potential Issues}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{State Sync Poisoning} (MEDIUM RISK)

  \begin{itemize}
  \tightlist
  \item
    Malicious peers could serve incorrect state
  \item
    Mitigated: Hash verification against committed root
  \item
    Consider: Peer reputation scoring for repeated failures
  \end{itemize}
\item
  \textbf{Memory Pressure} (LOW RISK)

  \begin{itemize}
  \tightlist
  \item
    Large state tries could exhaust memory
  \item
    Mitigated: Segmentation and worker limits
  \end{itemize}
\end{enumerate}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{func}\NormalTok{ NewEpoch}\OperatorTok{(}
\NormalTok{    upgrades upgrade}\OperatorTok{.}\NormalTok{Config}\OperatorTok{,}

\end{Shaded}

\textbf{Epoch Transition Logic}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Not activated before Granite: returns empty epoch
\item
  First epoch: uses parent P-Chain height as reference
\item
  Epoch sealed when parent issued after epoch end time
\item
  New epoch starts with incremented number
\end{enumerate}

\textbf{Security Observations}:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Clean Transition} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Epoch changes are deterministic
  \item
    Based on timestamp comparison against epoch end time
  \end{itemize}
\item
  \textbf{P-Chain Height Locking} (GOOD)

  \begin{itemize}
  \tightlist
  \item
    Each epoch locks to a specific P-Chain height
  \item
    Prevents validator set manipulation during epoch
  \end{itemize}
\end{enumerate}

\subsection{7. Conclusion}\label{7-conclusion}
