\subsubsection{{[}C-01{]} CollateralRegistry.recordBond() lacks access
control - Arbitrary bond
inflation}\label{c-01-collateralregistryrecordbond-lacks-access-control---arbitrary-bond-inflation}

\textbf{File:} CollateralRegistry.sol:266-278 \textbf{Severity:}
Critical \textbf{Impact:} Fund Loss

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{function recordBond(address token, uint256 amount) external \{}
\NormalTok{    // Note: Should add access control for only authorized bond contracts}
\NormalTok{    CollateralConfig storage config = collaterals[token];}
\NormalTok{    if (!config.whitelisted) revert NotWhitelisted(token);}
\NormalTok{    // ...}
\NormalTok{    config.totalBonded += amount;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} The \texttt{recordBond()} function has no access
control. Any address can call it to artificially inflate
\texttt{totalBonded}, potentially exhausting capacity limits and
blocking legitimate bonds.

\textbf{Attack Vector:}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Attacker calls \texttt{recordBond(token,\ maxCapacity)} repeatedly
\item
  \texttt{totalBonded} reaches \texttt{maxCapacity}
\item
  Legitimate users cannot bond (CapacityExceeded revert)
\item
  DoS on bonding mechanism
\end{enumerate}

\textbf{Recommendation:} Add access control modifier:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{mapping(address =\textgreater{} bool) public authorizedBonders;}

\NormalTok{function recordBond(address token, uint256 amount) external \{}
\NormalTok{    require(authorizedBonders[msg.sender], "Unauthorized");}
\NormalTok{    // ...}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\subsubsection{{[}C-02{]} Vault.init() one-time setup has no access
control - Permanent router
hijack}\label{c-02-vaultinit-one-time-setup-has-no-access-control---permanent-router-hijack}

\textbf{File:} Vault.sol:87-92 \textbf{Severity:} Critical
\textbf{Impact:} Fund Loss (permanent)

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{function init(address \_router) external \{}
\NormalTok{    if (router != address(0)) revert Invalid();}
\NormalTok{    if (\_router == address(0)) revert Zero();}
\NormalTok{    router = \_router;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} The \texttt{init()} function can be called by
anyone before the legitimate owner sets the router. An attacker can
front-run deployment and set a malicious router address.

\textbf{Attack Vector:}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Monitor mempool for Vault deployment
\item
  Front-run to call \texttt{init(attackerControlledRouter)}
\item
  All \texttt{flush()} and \texttt{flushAll()} operations now require
  attacker\textquotesingle s approval
\item
  Attacker can permanently block fee distribution
\end{enumerate}

\textbf{Recommendation:} Restrict to deployer or use constructor
initialization:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{constructor(address \_token, address \_router) \{}
\NormalTok{    token = IERC20(\_token);}
\NormalTok{    router = \_router;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\subsubsection{{[}C-03{]} LiquidBond oracle manipulation via stale price
- Arbitrage
attack}\label{c-03-liquidbond-oracle-manipulation-via-stale-price---arbitrage-attack}

\textbf{File:} LiquidBond.sol:362-388 \textbf{Severity:} Critical
\textbf{Impact:} Fund Loss

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{function \_getValueInSats(address token, uint256 amount) internal view returns (uint256) \{}
\NormalTok{    // ...}
\NormalTok{    (, int256 tokenPrice,,,) = IPriceFeed(config.priceFeed).latestRoundData();}
\NormalTok{    // No staleness check on updatedAt}
\NormalTok{    (, int256 btcPrice,,,) = IPriceFeed(btcPriceFeed).latestRoundData();}
\NormalTok{    // No staleness check}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} The oracle price fetching ignores
\texttt{updatedAt} timestamp. Stale prices enable arbitrage attacks.

\textbf{Attack Vector:}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Oracle stops updating (network congestion, oracle failure)
\item
  Real price moves significantly
\item
  Attacker bonds at favorable stale rate
\item
  Claims ASHA at discount far exceeding intended parameters
\end{enumerate}

\textbf{Recommendation:} Add staleness checks:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{uint256 public constant MAX\_PRICE\_STALENESS = 1 hours;}

\NormalTok{(, int256 tokenPrice,, uint256 updatedAt,) = IPriceFeed(config.priceFeed).latestRoundData();}
\NormalTok{if (block.timestamp {-} updatedAt \textgreater{} MAX\_PRICE\_STALENESS) revert StalePrice();}
\end{Highlighting}
\end{Shaded}

\subsubsection{{[}C-04{]} ProtocolLiquidity LP valuation vulnerable to
manipulation}\label{c-04-protocolliquidity-lp-valuation-vulnerable-to-manipulation}

\textbf{File:} ProtocolLiquidity.sol:463-480 \textbf{Severity:} Critical
\textbf{Impact:} Fund Loss

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{function \_getLPValue(address lpToken, uint256 lpAmount) internal view returns (uint256) \{}
\NormalTok{    (uint112 reserve0, uint112 reserve1,) = pool.getReserves();}
\NormalTok{    uint256 totalSupply = pool.totalSupply();}

\NormalTok{    uint256 price0 = oracle.getPrice(pool.token0());}
\NormalTok{    uint256 price1 = oracle.getPrice(pool.token1());}

\NormalTok{    uint256 value0 = uint256(reserve0) * price0 / 1e18;}
\NormalTok{    uint256 value1 = uint256(reserve1) * price1 / 1e18;}
\NormalTok{    uint256 totalPoolValue = value0 + value1;}

\NormalTok{    return (totalPoolValue * lpAmount) / totalSupply;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} LP token valuation using spot reserves is
manipulable via flash loans. Attacker can inflate reserves, bond LP, and
extract value.

\textbf{Attack Vector:}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Flash loan large amounts of both tokens
\item
  Add liquidity to inflate reserves
\item
  Call \texttt{bondLP()} at inflated LP valuation
\item
  Receive excessive ASHA
\item
  Remove liquidity and repay flash loan
\item
  Profit from arbitrage
\end{enumerate}

\textbf{Recommendation:} Use TWAP oracles or fair LP pricing:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{// Fair LP price = 2 * sqrt(reserve0 * reserve1) * sqrt(price0 * price1) / totalSupply}
\NormalTok{function \_getLPValueFair(address lpToken, uint256 lpAmount) internal view returns (uint256) \{}
\NormalTok{    (uint112 r0, uint112 r1,) = ILiquidityPool(lpToken).getReserves();}
\NormalTok{    uint256 sqrtK = Math.sqrt(uint256(r0) * uint256(r1));}
\NormalTok{    uint256 p0 = oracle.getPrice(pool.token0());}
\NormalTok{    uint256 p1 = oracle.getPrice(pool.token1());}
\NormalTok{    uint256 sqrtP = Math.sqrt(p0 * p1);}
\NormalTok{    return (2 * sqrtK * sqrtP * lpAmount) / (pool.totalSupply() * 1e18);}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\subsubsection{{[}C-06{]} ValidatorVault reward distribution precision
loss}\label{c-06-validatorvault-reward-distribution-precision-loss}

\textbf{File:} ValidatorVault.sol:134-148 \textbf{Severity:} Critical
\textbf{Impact:} Fund Loss (dust accumulation)

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{function \_distributeRewards(uint256 amount) internal \{}
\NormalTok{    if (amount == 0 || totalDelegated == 0) return;}

\NormalTok{    uint256 toReserve = (amount * slashingReserveBps) / BPS;}
\NormalTok{    slashingReserve += toReserve;}

\NormalTok{    uint256 toDistribute = amount {-} toReserve;}

\NormalTok{    // Precision loss: if toDistribute \textless{} totalDelegated, this rounds to 0}
\NormalTok{    accRewardPerShare += (toDistribute * 1e18) / totalDelegated;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} When
\texttt{toDistribute\ *\ 1e18\ \textless{}\ totalDelegated}, the
division rounds to zero. Small reward amounts become permanently stuck.

\textbf{Attack Vector:}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Delegator stakes very large amount (e.g., 1e30)
\item
  Small rewards (\textless{} 1e12) contribute 0 to accRewardPerShare
\item
  These funds remain in contract, never distributed
\item
  Over time, significant dust accumulation
\end{enumerate}

\textbf{Recommendation:} Track undistributed dust:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{uint256 public undistributedRewards;}

\NormalTok{function \_distributeRewards(uint256 amount) internal \{}
\NormalTok{    // ...}
\NormalTok{    uint256 distributedPerShare = (toDistribute * 1e18) / totalDelegated;}
\NormalTok{    uint256 actualDistributed = (distributedPerShare * totalDelegated) / 1e18;}
\NormalTok{    undistributedRewards += toDistribute {-} actualDistributed;}
\NormalTok{    accRewardPerShare += distributedPerShare;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\subsubsection{{[}H-01{]} Bond.sol token calculation can overflow with
extreme
parameters}\label{h-01-bondsol-token-calculation-can-overflow-with-extreme-parameters}

\textbf{File:} Bond.sol:142 \textbf{Severity:} High \textbf{Impact:}
Incorrect token allocation

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{uint256 tokensOwed = (amount * bond.tokensToMint * (10000 + bond.discount)) / (bond.targetRaise * 10000);}
\end{Highlighting}
\end{Shaded}

\textbf{Description:} With large \texttt{amount}, \texttt{tokensToMint},
and \texttt{discount}, the multiplication can exceed uint256.

\textbf{PoC:}

\begin{itemize}
\tightlist
\item
  amount = 1e24 (1M tokens with 18 decimals)
\item
  tokensToMint = 1e26
\item
  discount = 9000 (90\%)
\item
  Multiplication: 1e24 * 1e26 * 19000 = 1.9e54 (overflow)
\end{itemize}

\textbf{Recommendation:} Use intermediate calculations or mulDiv:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{uint256 discountMultiplier = 10000 + bond.discount;}
\NormalTok{uint256 tokensOwed = (amount * discountMultiplier).mulDiv(bond.tokensToMint, bond.targetRaise * 10000);}
\end{Highlighting}
\end{Shaded}

require weights sum validation before distribution.
