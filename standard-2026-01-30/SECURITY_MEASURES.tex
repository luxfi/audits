\begin{Shaded}

\NormalTok{    // Use Warp precompile for actual verification}

\NormalTok{        abi.encodeWithSignature("verifyMessage(bytes)", proof)}

\subsubsection{3. LP Valuation Flash-Loan Manipulation (CRITICAL -
Treasury
C-04)}\label{3-lp-valuation-flash-loan-manipulation-critical---treasury-c-04}

\textbf{Issue:} ProtocolLiquidity uses spot reserves for LP valuation.

\textbf{Impact:} Attackers can inflate LP value and extract excess ASHA.

\textbf{Remediation:}

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{// Use fair LP pricing formula (geometric mean)}
\NormalTok{function \_getLPValue(address lpToken, uint256 amount) internal view returns (uint256) \{}
\NormalTok{    (uint112 reserve0, uint112 reserve1,) = IUniswapV2Pair(lpToken).getReserves();}
\NormalTok{    uint256 totalSupply = IERC20(lpToken).totalSupply();}

\NormalTok{    // Fair value = 2 * sqrt(reserve0 * reserve1) * amount / totalSupply}
\NormalTok{    uint256 k = uint256(reserve0) * uint256(reserve1);}
\NormalTok{    uint256 fairReserve = 2 * Math.sqrt(k);}
\NormalTok{    return (fairReserve * amount) / totalSupply;}
\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\subsection{Security Defaults \& Best
Practices}\label{security-defaults--best-practices}

\subsubsection{1. Access Control}\label{1-access-control}

\begin{itemize}
\tightlist
\item
  All admin functions protected by \texttt{onlyOwner} or role-based
  access
\item
  Multi-sig recommended for production deployments
\item
  Timelock contracts for parameter changes
\end{itemize}

\subsubsection{2. Reentrancy Protection}\label{2-reentrancy-protection}

\begin{itemize}
\tightlist
\item
  \texttt{ReentrancyGuard} on all external state-changing functions
\item
  Checks-Effects-Interactions pattern enforced
\item
  CEI applied to all token transfers
\end{itemize}

\subsubsection{3. Oracle Security}\label{3-oracle-security}

\begin{itemize}
\tightlist
\item
  Maximum staleness check (1 hour default)
\item
  Price sanity bounds (±50\% from previous)
\item
  Multiple oracle fallback support
\end{itemize}

\subsubsection{4. Flash Loan Protection}\label{4-flash-loan-protection}

\begin{itemize}
\tightlist
\item
  Voting delay between proposal creation and voting start
\item
  Snapshot-based voting power (not live balance)
\item
  Block advancement required for delegation checkpoints
\end{itemize}

\subsubsection{5. Signature Security}\label{5-signature-security}

\begin{itemize}
\tightlist
\item
  OpenZeppelin ECDSA library for all signature verification
\item
  EIP-712 typed data for domain separation
\item
  Nonces for replay protection (monotonic counters)
\end{itemize}

\subsubsection{6. Gas Limits}\label{6-gas-limits}

\begin{itemize}
\tightlist
\item
  Batch size limits on all loops (MAX\_BATCH\_SIZE = 50)
\item
  Per-user position limits (MAX\_POSITIONS = 100)
\item
  Pagination for view functions
\end{itemize}
